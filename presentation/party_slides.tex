            \item<2-> People are represented as vertices
            \vspace{1mm}
            \item<2-> An edge exists between two people if they are acquainted

    \end{itemize}

\section{Theorem 1}

\frame{
    \begin{theorem}
        Let $E$ and $F$ be the number of empty and full triangles respectively. 
        Then in every graph with $p$ vertices
        % $$ E + F \geq \binom{p}{3} - \bfloor{ \frac{p}{2} \bfloor{ \bparen{ \frac{p-1}{2} }^2 } } $$
        \begin{equation} \label{eq:theorem1}
            E + F \geq \binom{p}{3} - \bfloor{ \frac{p}{2} \bfloor{ \bparen{ \frac{p-1}{2} }^2 } }
        \end{equation}
        and this lower bound is sharp for each positive integer $p$.
    \end{theorem}
}

\frame{ \frametitle{Proof}
    \begin{itemize}
        \item<1-> Let $P$ be the number of partial triangles in $G$, meaning the number of triangles containing 
        exactly one or two edges.
        \vspace{3mm}
        \item<2-> It's clear that
            \begin{equation} \label{eq:total_triangles}
                E + F + P = \binom{p}{3}.
            \end{equation}
    \end{itemize}    
}

\frame{ \frametitle{Proof}
    \begin{itemize}
        \item<1-> Let $d_i$ be the degree of a vertex $v_i$, in other words, the number of people acquainted with 
        the $i^{th}$ person.
        \vspace{3mm}
        \item<2-> For each vertex $v_i$, picking one of their $d_i$ acquaintances and one of their $p - 1 - d_i$ 
        nonacquaintances produces a partial triangle.
        \vspace{3mm}
        \item<3-> Thus, each $v_i$ produces $d_i(p - 1 - d_i)$ partial triangles.
    \end{itemize}
}

\frame{ \frametitle{Proof}
    \begin{itemize}
        \item<1-> Furthermore, we note that every partial triangle is counted twice in this manner.
        \vspace{3mm}
        \item<2-> To show this is true, let vertex $a$, $b$, and $c$ represent $v_i$, one of their acquaintances, and one of 
        their nonacquaintances respectively.
        \vspace{3mm}
        \item<3-> Producing a partial triangle from $v_i$ yields two cases.
    \end{itemize}
}

\frame{ \frametitle{Proof}
    \begin{itemize}
        \item<1-> Case 1: $b$ and $c$ are not acquainted.
        \vspace{1mm}
            \begin{figure}
                \centering
                \includegraphics[scale=.5]{../figures/partial_case_1.pdf}
            \end{figure}
        \vspace{3mm}
        \item<2-> Then this partial triangle is counted for $a$ and $b$.
    \end{itemize}
}

\frame{ \frametitle{Proof}
    \begin{itemize}
        \item<1-> Case 2: $b$ and $c$ are acquainted.
        \vspace{1mm}
            \begin{figure}
                \centering
                \includegraphics[scale=.5]{../figures/partial_case_2.pdf}
            \end{figure}
        \vspace{3mm}
        \item<2-> Then this partial triangle is counted for $a$ and $c$.
    \end{itemize}
}

\frame{ \frametitle{Proof}
    \begin{itemize}
        \item<1-> In both cases, the partial triangle is counted twice, which follows for 
        every partial triangle.
        \vspace{3mm}
        \item<2-> Consequently,
            \begin{equation} \label{eq:partial_triangles}
                P = \frac{1}{2} \sum^{p}_{i=1}{d_i(p - 1 - d_i)}.
            \end{equation}
    \end{itemize}
     
}

\frame{ \frametitle{Proof}
    \begin{itemize}
        \item<1-> Looking back at Equation (\ref{eq:total_triangles}), we can minimize $E + F$ by maximizing $P$.
        \vspace{3mm}
        \item<2-> To maximize $P$, we must maximize the sum in Equation (\ref{eq:partial_triangles}).
        \vspace{3mm}
            \begin{align*}
                E + F + P &= \binom{p}{3} & P &= \frac{1}{2} \sum^{p}_{i=1}{d_i(p - 1 - d_i)} \notag
            \end{align*}
    \end{itemize}
}

\frame{ \frametitle{Proof}
    \begin{equation}
        P = \frac{1}{2} \sum^{p}_{i=1}{d_i(p - 1 - d_i)} \notag
    \end{equation}
    \begin{itemize}
        \item<1-> We can view each term of the sum as a quadratic function of $d_i$.
        \vspace{3mm}
        \item<2-> After finding the derivative of the equation it's clear that we attain our maximum when $d_i = \frac{p - 1}{2}$. 
    \end{itemize}
}

\frame{ \frametitle{Proof}
    \begin{itemize}
        \item<1-> However, this is a contradiction when $p$ is even since $d_i$ is an integer.
        \vspace{3mm}
        \item<2-> Thus, if $p$ is odd, we attain the maximum value of $\frac{(p - 1)^2}{4}$ for 
        each term when $d_i = \frac{p - 1}{2}$.
        \vspace{3mm}
        \item<3-> If $p$ is even, we attain the maximum possible value of $\frac{p(p - 2)}{4}$ for 
        each term when $d_i = \frac{p}{2}$ or $d_i = \frac{p - 2}{2}$.
    \end{itemize}
}

\frame{ \frametitle{Proof}
    \begin{itemize}
        \item<1-> In either case, we can express the maximum value as $$ \bfloor{ \bparen{ \frac{p - 1}{2} }^2 }, $$
        \item<2-> and so
        \begin{equation} \label{eq:partial_max}
            P \leq \frac{1}{2} \sum^{p}_{i=1}{ \bfloor{ \bparen{ \frac{p - 1}{2} }^2 } } 
                = \frac{p}{2} \bfloor{ \bparen{ \frac{p - 1}{2} }^2 }.
        \end{equation}
    \end{itemize}
}

\frame{ \frametitle{Proof}
    \begin{itemize}
        \item<1-> But since $P$ is an integer, we can strengthen this to read
        \begin{equation} \label{eq:p_max}
            P \leq \bfloor{\frac{p}{2} \bfloor{ \bparen{ \frac{p - 1}{2} }^2 }}.
        \end{equation}
        \vspace{3mm}
        \item<2-> Equations (\ref{eq:total_triangles}) and (\ref{eq:p_max}) now yield our desired bound:
        \begin{equation} \label{eq:bound}
            E + F = \binom{p}{3} - P \geq \binom{p}{3} - \bfloor{\frac{p}{2} \bfloor{ \bparen{ \frac{p - 1}{2} }^2 }}.
        \end{equation}
    \end{itemize}
}

\frame{ \frametitle{Proof}
    \begin{itemize}
        \item<1-> Next, for each $p$, we must find a graph $G_p$ attaining this bound.
        \vspace{3mm}
        \item<2-> But equality in Equation (\ref{eq:theorem1}) is equivalent to equality in Equation (\ref{eq:p_max}).
        \vspace{3mm}
        \item<3-> Which occurs only when 
        \begin{equation} \label{eq:max_desired}
            P = \frac{1}{2} \sum^{p}_{i=1}{d_i(p - 1 - d_i)} 
                = \bfloor{\frac{p}{2} \bfloor{ \bparen{ \frac{p - 1}{2} }^2 }}.
        \end{equation}
    \end{itemize}
     
}

\frame{ \frametitle{Proof}
    \begin{itemize}
        \item<1-> If $p=2n$ for some integer $n$, let $G_p$ be the complete bipartite graph $K_{n,n}$.
            \begin{figure}
                \centering
                \includegraphics[scale=.3]{../figures/g_8.pdf}
                \caption{A visualization of $G_8$}
            \end{figure}
        \vspace{3mm}
        \item<2-> Now $G_p$ is regular of degree $n$, and we must check that Equation (\ref{eq:max_desired}) is satisfied.
    \end{itemize}
}

\frame{ \frametitle{Proof}
    \scriptsize{
        \begin{align*}
            P &= \frac{1}{2} \sum^{p}_{i=1}{d_i(p - 1 - d_i)} & P &= \bfloor{\frac{p}{2} \bfloor{ \bparen{ \frac{p - 1}{2} }^2 }} \notag \\
            &= \frac{1}{2} \sum^{2n}_{i=1}{n(2n - 1 - n)}   &   &= \bfloor{\frac{2n}{2} \bfloor{ \bparen{ \frac{2n - 1}{2} }^2 }} \notag \\
            &= \frac{1}{2} \sum^{2n}_{i=1}{n^2 - n}         &   &= \bfloor{ n \bfloor{ \frac{4n^2 - 4n + 1}{4} }} \notag \\
            &= \frac{1}{2} (2n^3 - 2n^2)                    &   &= \bfloor{ n \bfloor{ n^2 - n + \frac{1}{4} }} \notag \\
            &= n^3 - n^2                                    &   &= \bfloor{ n(n^2 - n) }  \notag \\
            &                                               &   &= \bfloor{ n^3 - n^2 } \notag \\
            &                                               &   &= n^3 - n^2 \notag
        \end{align*}
    }
}

\frame{ \frametitle{Proof}
    \begin{itemize}
        \item<1-> We can see that Equation (\ref{eq:max_desired}) holds when $p=2n$.
        \vspace{3mm}
        \item<2-> If $p=2n+1$, the construction of $G_p$ is a bit more involved.
        \vspace{3mm}
        \item<3-> We will walk through the construction of $G_9$ as an example.
    \end{itemize}
}

\frame{ \frametitle{Proof}
    \begin{itemize}
        \item We start with $K_{n,n}$ with its vertices labeled $u_1, u_2,...,u_n$; $v_1, v_2,...,v_n$.
    \end{itemize}

    \begin{figure}
        \centering
        \includegraphics[scale=.3]{../figures/construction_step_1.pdf}
    \end{figure}
}

\frame{ \frametitle{Proof}
    \begin{itemize}
        \item We subdivide edge $u_iv_i$ for $i \leq \frac{n}{2}$.
    \end{itemize}

    \begin{figure}
        \centering
        \includegraphics[scale=.3]{../figures/construction_step_2.pdf}
    \end{figure}
}

\frame{ \frametitle{Proof}
    \begin{itemize}
        \item We now obtain $G_p$ by using these $\floor{\frac{n}{2}}$ 
        subdivision points to form a single vertex labeled $w$.
    \end{itemize}

    \begin{figure}
        \centering
        \includegraphics[scale=.3]{../figures/construction_of_g_p.pdf}
    \end{figure}
}

\frame{ \frametitle{Proof}
    \begin{itemize}
        \item We observe that $G_p$ has $2n$ vertices of degree $n$ and 
        one vertex of degree $2\floor{\frac{n}{2}}$.
    \end{itemize}

    \begin{figure}
        \centering
        \includegraphics[scale=.3]{../figures/construction_of_g_p.pdf}
    \end{figure}
}

\frame{ \frametitle{Proof}
    \begin{itemize}
        \item We observe that $G_p$ has $2n$ vertices of degree $n$ and 
        one vertex of degree $2\floor{\frac{n}{2}}$.
        \vspace{3mm}
        \item It is routine to check that Equation (\ref{eq:max_desired}) is satisfied.
    \end{itemize}
}

\frame{ \frametitle{Proof}
    \scriptsize{
        \begin{align*}
            P &= \frac{1}{2} \sum^{p}_{i=1}{d_i(p - 1 - d_i)}                                                    & P &= \bfloor{\frac{p}{2} \bfloor{ \bparen{ \frac{p - 1}{2} }^2 }} \notag \\
            &= \frac{1}{2} \sum^{2n+1}_{i=1}{d_i(2n - d_i)}                                                      &   &= \bfloor{\frac{2n+1}{2} \bfloor{ \bparen{ \frac{2n + 1 - 1}{2} }^2 }} \notag \\
            &= \frac{1}{2} \bparen{\sum^{2n}_{i=1}{n(2n - n)} + 2\floor{\frac{n}{2}}(2n - 2\floor{\frac{n}{2}})} &   &= \bfloor{\frac{2n+1}{2} \bfloor{ n^2 }} \notag \\
            &= \frac{1}{2} \bparen{\sum^{2n}_{i=1}{n^2} + 2\floor{\frac{n}{2}}(2n - 2\floor{\frac{n}{2}})}       &   &= \bfloor{\bparen{\frac{2n+1}{2}} n^2} \notag \\
            &= \frac{1}{2} \bparen{2n^3 + 2\floor{\frac{n}{2}}(2n - 2\floor{\frac{n}{2}})}                       &   &= \bfloor{\frac{2n^3+n^2}{2}}  \notag \\
            &                                                                                                    &   &= \bfloor{ n^3 + \frac{1}{2}n^2 } \notag
        \end{align*}
    }
}

\frame{ \frametitle{Proof}
    \begin{itemize}
        \item At this point, we run into two cases.
            \begin{align*}
                P &= \frac{1}{2} \bparen{2n^3 + 2\floor{\frac{n}{2}}(2n - 2\floor{\frac{n}{2}})} & P &= \bfloor{ n^3 + \frac{1}{2}n^2 } \notag
            \end{align*}
    \end{itemize}        
}

\frame{ \frametitle{Proof}
    \begin{itemize}
        \item Case 1: $n$ is even. Let $n = 2t$ for some integer $t$.
    \end{itemize}

    \small{
        \begin{align*}
            P &= \frac{1}{2} \bparen{2n^3 + 2\floor{\frac{n}{2}}(2n - 2\floor{\frac{n}{2}})}         & P &= \bfloor{ n^3 + \frac{1}{2}n^2 } \notag \\
            &= \frac{1}{2} \bparen{2(2t)^3 + 2\floor{\frac{2t}{2}}(2(2t) - 2\floor{\frac{2t}{2}})} &   &= \bfloor{ (2t)^3 + \frac{1}{2}(2t)^2 } \notag \\
            &= \frac{1}{2} \bparen{16t^3 + 2t(4t - 2t)}                                            &   &= \bfloor{ 8t^3 + \frac{1}{2}4t^2 } \notag \\
            &= \frac{1}{2} \bparen{16t^3 + 4t^2}                                                   &   &= \bfloor{ 8t^3 + 2t^2 } \notag \\
            &= 8t^3 + 2t^2                                                                         &   &= 8t^3 + 2t^2 \notag
        \end{align*}
    }
}

\frame{ \frametitle{Proof}
    \begin{itemize}
        \item Case 2: $n$ is odd. Let $n = 2t + 1$ for some integer $t$.
    \end{itemize}

    \scriptsize{
        \begin{align}
            P &= \frac{1}{2} \bparen{2n^3 + 2\floor{\frac{n}{2}}(2n - 2\floor{\frac{n}{2}})}                                \notag \\
            &= \frac{1}{2} \bparen{2(2t + 1)^3 + 2\bfloor{\frac{2t + 1}{2}}(2(2t + 1) - 2\bfloor{\frac{2t + 1}{2}})}        \notag \\
            &= \frac{1}{2} \bparen{2(8t^3 + 12t^2 + 6t + 1) + 2\floor{t + \frac{1}{2}}(4t + 2 - 2\floor{t + \frac{1}{2}})}  \notag \\
            &= \frac{1}{2} \bparen{16t^3 + 24t^2 + 12t + 2 + 2t(4t + 2 - 2t)}                                               \notag \\
            &= \frac{1}{2} \bparen{16t^3 + 28t^2 + 16t + 2}                                                                 \notag \\
            &= 8t^3 + 14t^2 + 8t + 1                                                                                        \notag
        \end{align}
    }
}

\frame{ \frametitle{Proof}
    \begin{itemize}
        \item Case 2: $n$ is odd. Let $n = 2t + 1$ for some integer $t$.
    \end{itemize}

    \scriptsize{
        \begin{align}
            P &= \bfloor{ n^3 + \frac{1}{2}n^2 }                                                                            \notag \\
            &= \bfloor{ (2t + 1)^3 + \frac{1}{2}(2t + 1)^2 }                                                                \notag \\
            &= \bfloor{ 8t^3 + 12t^2 + 6t + 1 + \frac{4t^2 + 4t + 1}{2} }                                                   \notag \\
            &= \bfloor{ 8t^3 + 12t^2 + 6t + 1 + 2t^2 + 2t + \frac{1}{2} }                                                   \notag \\
            &= \bfloor{ 8t^3 + 14t^2 + 8t + 1 + \frac{1}{2} }                                                               \notag \\
            &= 8t^3 + 14t^2 + 8t + 1                                                                                        \notag
        \end{align}
    }
}

\frame{ \frametitle{Proof}
    \begin{itemize}
        \item Thus, in both cases, Equation (\ref{eq:max_desired}) holds. $\square$
    \end{itemize}
}
